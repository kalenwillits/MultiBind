# MultiBind Negate Operator (~) Test Configuration
# Test the ~ negate operator for exclusive button combinations

# ========== BASIC NEGATION TESTS ==========
# Single negated button
~000*010=sim/test/not_000_held_010    # Only triggers if 000 NOT held and 010 held
~001*011=sim/test/not_001_held_011    # Only triggers if 001 NOT held and 011 held

# ========== MULTIPLE NEGATION TESTS ==========
# Multiple negated buttons
~000~001*020=sim/test/not_000_not_001_held_020    # 000 NOT held, 001 NOT held, 020 held

# ========== MIXED PATTERN TESTS ==========
# Combination of negated and regular triggers
*005~010*015=sim/test/005_not_010_015            # 005 held, 010 NOT held, 015 held
~000*010+011=sim/test/not_000_010_then_011       # 000 NOT held, 010 held, then 011 pressed

# ========== AXIS BINDING WITH NEGATION ==========
# Negation with axis bindings - continuous control
~000*010^A04=sim/engines/throttle_1              # 000 NOT held, 010 held → throttle control
~001*020^A08=sim/engines/prop_1                  # 001 NOT held, 020 held → prop control
~000~001*030^A09=sim/engines/mixture_1           # 000 NOT held, 001 NOT held, 030 held → mixture

# ========== EDGE CASES ==========
# Negation at different positions in pattern
~005*010*015=sim/test/negated_first              # Negation first
*005~010*015=sim/test/negated_middle             # Negation in middle
*005*010~015=sim/test/negated_last               # Negation last (note: ~015 means "015 NOT held", still need pattern trigger)

# ========== PRACTICAL USE CASES ==========
# Prevent accidental trigger when another button is held
~050*051=sim/lights/landing_lights_toggle        # Landing lights only if safety (050) NOT engaged
~050*052=sim/gear/landing_gear_toggle            # Gear toggle only if safety NOT engaged

# Mode-exclusive controls
~100*110=sim/autopilot/mode_a                    # Mode A only if mode selector (100) NOT active
*100*110=sim/autopilot/mode_b                    # Mode B only if mode selector (100) IS active

# ========== BACKWARD COMPATIBILITY TESTS ==========
# Ensure regular patterns still work
*030+031=sim/electrical/battery_on               # Regular pattern (no negation)
+005-006=sim/lights/landing_lights_toggle        # Regular pattern (no negation)

# ========== STRESS TESTS ==========
# Complex negation patterns
~000~001~002*050*051*052=sim/test/complex_negation    # Multiple negations + multiple positives
~010*020^A04=sim/test/negation_axis_stress            # Negation with continuous axis binding
